\usepackage{etoolbox}

% Add space before citations.
\LetLtxMacro\oldcitep\citep
\RenewDocumentCommand\citep{o o m}{%
  \>%
  \IfNoValueTF{#1}{%
    \oldcitep{#3}%
  }{%
    \IfNoValueTF{#2}{%
      \oldcitep[#1]{#3}%
    }{%
      \oldcitep[#1][#2]{#3}%
    }%
  }%
}
\let\origcitetext\citetext
\renewcommand{\citetext}[1]{\>\origcitetext{#1}}
